\begin{myslidefragile}{b}{References with Bib\LaTeX}

\lstinline{\cite} with an example: \cite{Shampine2007}

\lstinline{\textcite} with an example: \textcite{Shampine2007}

\lstinline{\parencite} with an example: \parencite{Shampine2007}

\lstinline{\fullcite} with an example: \fullcite{Shampine2007}

\vspace*{\baselineskip}

\lstinline{\citetitle} with an example: \citetitle{Shampine2007}

\lstinline{\citetitle} with an example: \citetitle{Martins2021}

\lstinline{\citeauthor} with an example: \citeauthor{Shampine2007}

\lstinline{\citeurl} with an example: \citeurl{Margalit2017}

\vspace*{\baselineskip}

Multiple citations work like this example \parencite{Martins2021, Boyd2009, Cipra2000} and in a footnote\parnote{\cite{Martins2021, Boyd2009, Cipra2000}}

\vspace*{\baselineskip}

See command \lstinline{\refslides} for printing the references

\end{myslidefragile}
